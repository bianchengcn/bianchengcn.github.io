demo1、分支语句示例（if-else语句）：

#include <stdio.h>

int main() {
    int num = 10;
    
    if (num > 0) {
        printf("The number is positive.\n");
    } else if (num < 0) {
        printf("The number is negative.\n");
    } else {
        printf("The number is zero.\n");
    }
    
    return 0;
}

demo2、分支语句示例（switch语句）：

#include <stdio.h>

int main() {
    int choice;
    
    printf("Enter a number (1-3): ");
    scanf("%d", &choice);
    
    switch (choice) {
        case 1:
            printf("You selected option 1.\n");
            break;
        case 2:
            printf("You selected option 2.\n");
            break;
        case 3:
            printf("You selected option 3.\n");
            break;
        default:
            printf("Invalid choice.\n");
    }
    
    return 0;
}

demo3、循环语句示例（for循环）：

#include <stdio.h>

int main() {
    int i;
    
    for (i = 1; i <= 5; i++) {
        printf("%d\n", i);
    }
    
    return 0;
}

demo4、循环语句示例（while循环）：

#include <stdio.h>

int main() {
    int i = 1;
    
    while (i <= 5) {
        printf("%d\n", i);
        i++;
    }
    
    return 0;
}

demo5、循环语句示例（do-while循环）：

#include <stdio.h>

int main() {
    int i = 1;
    
    do {
        printf("%d\n", i);
        i++;
    } while (i <= 5);
    
    return 0;
}

demo6、嵌套分支语句示例：

#include <stdio.h>

int main() {
    int num = 10;

    if (num > 0) {
        if (num < 5) {
            printf("The number is between 0 and 5.\n");
        } else {
            printf("The number is greater than 5.\n");
        }
    } else {
        printf("The number is less than or equal to 0.\n");
    }

    return 0;
}

demo7、循环控制语句示例（break和continue）：

#include <stdio.h>

int main() {
    int i;

    for (i = 1; i <= 5; i++) {
        if (i == 3) {
            continue; // 跳过本次循环，继续下一次循环
        }

        if (i == 5) {
            break; // 结束循环
        }

        printf("%d\n", i);
    }

    return 0;
}

js树形递归获取树形里面的id值

let tree = []
function findDep(node = []) {
  for (let item of node) {
    tree.push(item.id)
    if (item.children) {
      findDep(item.children)
    }
  }
}
findDep([{id: 1, name: 'arr'}])

C++拯救007

1.题目描述

在老电影“007之生死关头”（Live and Let Die）中有一个情节，007被毒贩抓到一个鳄鱼池中心的小岛上，他用了一种极为大胆的方法逃脱 —— 直接踩着池子里一系列鳄鱼的大脑袋跳上岸去！（据说当年替身演员被最后一条鳄鱼咬住了脚，幸好穿的是特别加厚的靴子才逃过一劫。）

设鳄鱼池是长宽为100米的方形，中心坐标为 (0, 0)，且东北角坐标为 (50, 50)。池心岛是以 (0, 0) 为圆心、直径15米的圆。给定池中分布的鳄鱼的坐标、以及007一次能跳跃的最大距离，你需要告诉他是否有可能逃出生天。

2.解题思路

根据数据结构中图的思想，把每一个鳄鱼及岸边视为结点，采用深度优先搜索的方法，判断能否到下一个结点。如果从该节点可以跳到岸边，退出循环，是安全的。

深度优先遍历时，对每一个访问的结点设置该节点已经访问过，并判断从该节点是否可以一次跳到岸边，可以的话直接返回，否则，对其他结点判断，如果该节点可以到达且没有访问过，就对这个结点进行深度优先遍历。

3.代码实现

#include "iostream"
#include "algorithm"
using namespace std;

#define Yes 1
#define No 0
#define Max 100 //the maxnum of the crocodile
int N;// the number of crocodile
double D;//步长
bool answer;
//position ofcrocodile
struct Point
{
	double x;
	double y;
}Point[Max];

//the sign of visited
int Visited[Max];
void ResetVisited()
{
	for (int i = 0; i < N; i++)
		Visited[i] = 0;
}

//判断第一次能否跳成功
bool FirstJump(int i)
{
	if (pow(D + 7.5, 2) >= pow(Point[i].x, 2) + pow(Point[i].y, 2))
		return true;
	else 
		return false;
}

//判断每次能否跳成功
bool Jump(int i, int j)
{
	if (pow(D, 2) >= (pow(Point[i].x - Point[j].x, 2) + pow(Point[i].y - Point[j].y, 2)))
		return true;
	else
		return false;
}

//判断该点是否安全
bool IsSafe(int i)
{
	double xx = Point[i].x;
	double yy = Point[i].y;
	if (xx < 0)
		xx = -xx;
	if (yy < 0)
		yy = -yy;
	if (D >= 50 - xx || D >= 50 - yy)
		return true;
	else
		return false;
}

//DFS算法
bool DFS(int i)
{
	Visited[i] = 1;//记录遍历的痕迹
	if (IsSafe(i)) answer = Yes;
	else
	{
		answer = No;
		for (int j = 0; j < N;j++ )
		{
			//如果该节点未被访问，且能从i结点跳至j结点
			if (!Visited[j] && Jump(i, j))
			{
				answer = DFS(j);
			}
			if (answer) break;
		}
	}
	return answer;
}

//能否救出的判断
void Save007()
{
	void ResetVisited();
	for (int ii = 0; ii < N; ii++)
	{
		//如果该节点未被访问，且能从岛上跳过去
		if (!Visited[ii] && FirstJump(ii))
		{
			answer = DFS(ii);
			if (answer) break;
		}
	}
	if (answer) cout << "YES!" << endl;
	else cout << "NO!" << endl;
}

int main()
{
	cout << "请输入鳄鱼数目：";
	cin >> N;
	cout << "请输入步长：";
	cin >> D;
	for (int i = 0; i < N; i++)
	{
		cout << "\n请输入鳄鱼"<<i+1<<"的坐标：";
		cin >> Point[i].x;
		cin >> Point[i].y;
	}
	Save007();
	return 0;
}

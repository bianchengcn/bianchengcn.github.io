c++ 语言三角形判断

 

#include <iostream>

using namespace std;

int main()
{
    float a,b,c;
    cout << "请输入三角形的三条边:" << endl;
    cin>>a>>b>>c;
    if(a+b>c&&a+c>b&&b+c>a){
        
        if(a==b&&b==c){
                cout << "三条边可以构成一个等边三角形" << endl;
        }else if(a==b||a==c||b==a){
                cout << "三条边可以构成一个等腰三角形" << endl;
        }else{
                cout << "三条边可以构成一个三角形" << endl;
        }
    }else{
        cout << "三条边不可以构成一个三角形" << endl;
    }
    return 0;
}

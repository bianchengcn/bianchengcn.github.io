结构体变量的输入输出

#include<stdio.h>
int main(void)
{
struct student
{
char *name;
int num;
int age;
char sex;
float score;
}s1,s2;
s1.name="Zhang ping";
s1.num=102;
s1.age=23;
printf("input sex and score\n");
scanf("%c %f",&s1.sex,&s1.score);
s2=s1;
printf("Name=%s\nNumber=%d\nAge=%d\n",s2.name,s2.num,s2.age);
printf("Sex=%c\nScore=%f\n",s2.sex,s2.score);
return 0;
}

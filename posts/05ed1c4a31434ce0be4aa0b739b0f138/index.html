js中Object设值和取值

一、js中Object的设置和取值

将一个数组中的多个Object遍历取出key和value放入一个新Object中。
遍历数组获取Object值取新Object的value。

二、使用步骤

代码如下（示例）：

//从数组中取值放入object
let keywords = [{name:'first',value:'Hello'},
				{name:'second',value:'World'},
				{name:'third',value:'Good'},
				{name:'fourth',value:'Morning'}]
let obj = {};
for(let i =0 ;i < keywords.length;i++) {
	let kw = keywords[i];
	this.$set(obj,kw.name,kw.value);
}
console.log(obj);
/**
{
 first:'Hello',
 second:'World',
 third:'Good',
 fourth:'Morning'
}
**/
//循环从Object取值
for(let i =0 ;i < keywords.length;i++) {
	let kw = keywords[i];
	let val = obj[kw.name];
	console.log(val);//Helllo
}

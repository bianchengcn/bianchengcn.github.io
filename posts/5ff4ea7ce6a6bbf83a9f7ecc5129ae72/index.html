力扣：给你一个按 非递减顺序 排序的整数数组 nums，返回 每个数字的平方 组成的新数组，要求也按 非递减顺序 排序。

class Solution {
public:
    vector<int> sortedSquares(vector<int>& nums) {
       vector<int> square;
       for(int i=0;i<nums.size();i++){
           square.push_back(nums[i]*nums[i]);
       }
       //进行排序
       sort(square.begin(),square.end());
     return square;
    }
};

class Solution {
    public List<List<Integer>> fourSum(int[] nums, int target) {
        Arrays.sort(nums);
        List<List<Integer>> ret = new ArrayList<>();
        int n = nums.length;
        for(int a = 0;a<n;){
            for(int b = a+1;b<n;){
                int left = b+1;
                int right = n-1;
                long aim = (long)target- nums[a] - nums[b];

                while(left < right){

                    int sum = nums[left] + nums[right];
                    if(sum < aim){
                        left++;
                    }else if(sum > aim){
                        right--;
                    }else{
                        ret.add(Arrays.asList(nums[a] , nums[b] , nums[left++] , nums[right--]));
                        // ret.add(new ArrayList<Integer>(Arrays.asList(nums[left],nums[right],nums[i])));
                while(left < right && nums[left] == nums[left - 1]){
                    left++;
                }
                while(left < right && nums[right] == nums[right + 1]){
                    right--;
                }  
            }
        }

        b++;
        while(b < n && nums[b] == nums[b-1]){
            b++;
        }
    }
        a++;
        while(a < n && nums[a]==nums[a-1]){
            a++;
            }

        }
    return ret;
    }
}

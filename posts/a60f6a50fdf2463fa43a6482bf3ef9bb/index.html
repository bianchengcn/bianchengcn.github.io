这里是算得是半年和一年之后的日期

 var time = new Date('2021-12-17');
    console.log('当前年月' + time)
    console.log(time.getFullYear()+'-'+(time.getMonth()+1)+'-'+time.getDate())

    // 时间加一年
    time.setMonth(time.getMonth() + 12);
    // 或者
    time.setFullYear(time.getFullYear() + 1);
    console.log('一年后的年月' + time)
    console.log(time.getFullYear()+'-'+(time.getMonth()+1)+'-'+time.getDate())

    // 时间加半年
    time.setMonth(time.getMonth() + 6);
    console.log('半年后的时间' + time)

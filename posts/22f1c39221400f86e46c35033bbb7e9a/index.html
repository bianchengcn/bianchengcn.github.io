	function change(num){
		var changes = document.getElementsByName("schange");
		var arr = document.getElementsByName("name");
		if(num == 1){
			for(var i=0;i<arr.length;i++){
				arr[i].checked = true;
			}
		}else if(num==2){
			for(var i=0;i<arr.length;i++){
				arr[i].checked = false;
			}
		}else if(num==3){
			for(var i=0;i<arr.length;i++){
				if(arr[i].checked == true){
					arr[i].checked = false;
				}else if(arr[i].checked == false){
					arr[i].checked = true;
				}
			}
		}		
	}
</script>

删除字符串中的重复字符，只使用一个数组，尝试用指针实现。

程序：

#include <stdio.h>
int main(int argc, const char *argv[])
{
	char arr[30]="qwertyuwertasdfgwq";
	char *p=arr;
	int len=sizeof(arr)/sizeof(arr[0]);
	int i=0;
	int j=0;
	int k=0;
	
	for(i=0;i<len;i++)
	{
		char s=*(p+i);
		for(j=0;j<len;j++)
		{
			if(s==*(p+j) && i!=j)
			{
				for(k=j;k<len;k++)
				{
				*(p+k)=*(p+k+1);
				}
			}
		}
	}
	puts(p);
			
	return 0;
}

   $data = file_get_contents('php://input');

Think\Log::record('测试日志信息，这是警告级别','WARN',true);

采用record方法记录的日志信息不是实时保存的，如果需要实时记录的话，可以采用write方法，例如：

Think\Log::write('测试日志信息，这是警告级别，并且实时写入','WARN');

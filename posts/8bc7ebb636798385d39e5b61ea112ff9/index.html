Widget::Widget(QWidget * parent)
	: QDialog(parent),
	ui(new Ui::Widget)
{
	ui->setupUi(this);

    //去掉原widget窗口边框
	setWindowFlags(Qt::MSWindowsFixedSizeDialogHint);

	//初始化
	init();	
}
//初始化connect函数
void Widget::init()
{
	connectALL();
}
//connect槽函数编写
int Widget::connectALL()
 {
    connect(ui->pushButton, &QPushButton::clicked, [=]() {
		ui->plainTextEdit->clear();
    });

    return true;
}

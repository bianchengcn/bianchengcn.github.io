6 结构体

6.1 结构体基本概念

结构体属于用户自定义的数据类型，允许用户存储不同的数据类型。

6.2 结构体定义与使用

语法：

struct 结构体名{
    结构体成员列表
};

通过结构体创建变量的方式有三种：

  • struct 结构体名 变量名
  • struct 结构体名 变量名={成员1值，成员2值…}
  • 定义结构体时顺便创建变量

代码示例：

#include <iostream>
using namespace std;
#include <string>

//1、创建学生数据类型（姓名，年龄，分数）
struct Stu {
	string name;
	int age;
	double score;
}s3;

int main() {
	//2、通过学生类型创建具体学生（三种方式）
	struct Stu s1;//创建结构体变量，struct可省略
	//给s1赋值，通过.访问结构体变量中的属性
	s1.name = "张三";
	s1.age = 25;
	s1.score = 87.5;
	cout << "姓名：" << s1.name << " 年龄：" << s1.age << " 分数：" << s1.score << endl;
	cout << endl;
	struct Stu s2 = { "李四",26,92.8 };
	cout << "姓名：" << s2.name << " 年龄：" << s2.age << " 分数：" << s2.score << endl;
	cout << endl;
	s3.name = "王五";
	s3.age = 24;
	s3.score = 76.8;
	cout << "姓名：" << s3.name << " 年龄：" << s3.age << " 分数：" << s3.score << endl;
}

运行结果：

姓名：张三 年龄：25 分数：87.5

姓名：李四 年龄：26 分数：92.8

姓名：王五 年龄：24 分数：76.8

总结：

  • 定义结构体时的关键字时struct，不可省略
  • 创建结构体变量时，关键字structural可以省略
  • 结构体变量利用操作符” . “访问成员

struct声明尽量在外部声明（在main函数前面），不要在内部（main函数里）声明。因为外部声明可以被后面的任何函数使用，而内部声明只能被该声明所属函数使用。

6.3 结构体数组

作用： 将自定义的结构体放入到数组中方便维护。

语法：

struct 结构体名 数组名[元素个数]={
    {}，
    {}，
    ...    
}

代码示例：

#include <iostream>
using namespace std;
#include <string>

//创建学生数据类型（姓名，年龄，分数）
struct Stu {
	string name;
	int age;
	double score;
};

int main() {
	//创建结构体数组
	Stu arr[3] = {
		{"张三",25,87.5},
		{"李四",24,92.4},
	    {"王五",27,78.2}
	};
	//给结构体数组中的元素赋值
	arr[2].name = "赵六";
	arr[2].age = 35;
	arr[2].score = 90.1;
	//遍历结构体数组
	for (int i = 0; i < 3; i++) {
		cout << "姓名：" << arr[i].name << " 年龄：" << arr[i].age << " 分数：" << arr[i].score << endl;
	}
}

运行结果：

姓名：张三 年龄：25 分数：87.5
姓名：李四 年龄：24 分数：92.4
姓名：赵六 年龄：35 分数：90.1

6.4 结构体指针

作用： 通过指针访问结构体中的成员

  • 利用操作符->可以通过结构体指针访问结构体属性

代码示例：

#include <iostream>
using namespace std;
#include <string>

//结构体指针
struct stu {
	string name;
	int age;
	int score;
};
int main() {
	//创建结构体变量
	stu s = { "张三",25,78 };
	//通过指针指向结构体变量
	stu *p = &s;
	//通过指针访问结构体变量的数据
	cout << "姓名：" << p->name << " 年龄：" << p->age << " 分数：" << p->score << endl;
}

运行结果：

姓名：张三 年龄：25 分数：78

6.5 结构体嵌套结构体

作用： 结构体中的成员时另一个结构体。

例如： 每个老师辅导一个学员，一个老师的结构体中，记录一个学生的结构体。

代码示例：

#include <iostream>
using namespace std;
#include <string>
//结构体嵌套
struct Stu {
	string name;
	int age;
	int score;
};

struct Tea {
	int id;
	string name;
	int age;
	Stu stu;//教的学生
};
int main() {
	Tea t;
	//赋值
	t.id = 1000;
	t.age = 35;
	t.name = "老王";
	t.stu.name = "张三";
	t.stu.age = 23;
	t.stu.score = 87;
	//输出
	cout << "老师编号：" << t.id << " 老师姓名：" << t.name << " 老师年龄：" << t.age << " 老师所教的学生姓名："
		<< t.stu.name << " 老师所教的学生年龄：" << t.stu.age << " 老师所教的学生成绩：" << t.stu.score << endl;

}

运行结果：

老师编号：1000 老师姓名：老王 老师年龄：35 老师所教的学生姓名：张三 老师所教的学生年龄：23 老师所教的学生成绩：87

6.6 结构体做函数参数

作用： 将结构体作为参数向函数中传递

传递的方式有两种：

  • 值传递
  • 地址传递

代码示例：

#include <iostream>
using namespace std;
#include <string>

struct Stu {
	string name;
	int age;
	int score;
};
//值传递
void print_stu1(Stu stu) {
	stu.age = 100;//修改值
	cout << "子函数中 姓名：" << stu.name << " 年龄：" << stu.age << " 分数：" << stu.score << endl;
}

//地址传递
void print_stu2(Stu *p) {
	p->age = 200;//修改值
	cout << "子函数中 姓名：" << p->name<< " 年龄：" << p->age << " 分数：" << p->score << endl;
}
int main() {
	//创建结构体变量
	Stu stu = { "张三",25,88 };
	//调用函数
	print_stu1(stu);
	cout << "主函数中 姓名：" << stu.name << " 年龄：" << stu.age << " 分数：" << stu.score << endl;
	cout << endl;
	print_stu2(&stu);
	cout << "主函数中 姓名：" << stu.name << " 年龄：" << stu.age << " 分数：" << stu.score << endl;
}

运行结果：

子函数中 姓名：张三 年龄：100 分数：88
主函数中 姓名：张三 年龄：25 分数：88

子函数中 姓名：张三 年龄：200 分数：88
主函数中 姓名：张三 年龄：200 分数：88

注意：值传递时，如果形参发生改变，不会影响实参。

​ 地址传递中，形参改变，实参也会跟着改变。

6.7 结构体中const使用场景

作用： 用const来防止误操作。

代码示例：

#include <iostream>
using namespace std;
#include <string>

struct Stu {
	string name;
	int age;
	int score;
};
//将函数中的形参改为指针，可以减少内存空间，不会复制新的副本出来
//加入const后，不可修改，防止误操作
void print_stu(const Stu *p) {
	for (int i = 0; i < 3;i++) {
		cout << "姓名：" << p->name << " 年龄：" << p->age << " 分数：" << p->score << endl;
		p++;
	}
}
int main() {
	//创建结构体变量
	Stu stu[3] = {
		{"张三",25,87},
		{"李四",24,92},
		{"王五",27,78}
	};
	print_stu(&stu[0]);
}

运行结果：

姓名：张三 年龄：25 分数：87
姓名：李四 年龄：24 分数：92
姓名：王五 年龄：27 分数：78

6.8 结构体案例1

案例描述：

设计学生和老师的结构体，其中在老师的结构体中，有老师姓名和一个存放5名学生的数组作为成员。

学生的成员有姓名、考试分数，创建数组存放3名老师，通过函数给每个老师及所带的学生赋值，

最终打印出老师数据和老师所带的学生数据。

实现代码：

#include <iostream>
using namespace std;
#include <string>
#include <ctime>
/*
设计学生和老师的结构体，其中在老师的结构体中，有老师姓名和一个存放5名学生的数组作为成员。

学生的成员有姓名、考试分数，创建数组存放3名老师，通过函数给每个老师及所带的学生赋值，

最终打印出老师数据和老师所带的学生数据。*/

//学生结构体
struct Stu {
	string stu_name;
	int score;
};

//老师结构体
struct Tea {
	string tea_name;
	Stu stu[5];
};
//赋值
void full(Tea* p,int m,int n) {
	string name[] = { "stu_a","stu_b","stu_c","stu_d","stu_e" };
	string teacher[] = { "tea_a","tea_b","tea_c" };
	srand((unsigned int)time(NULL));
	Tea* q = p;
	int i = 0;
	for (q,i; q != p + n; q++,i++) {
		q->tea_name = teacher[i];
		for (int j=0; j < m; j++) {
			q->stu[j].stu_name= name[j];
			q->stu[j].score= rand() % 100 + 1;
		}
	}
}
//输出
void print(Tea* p, int m, int n) {
	Tea* q = p;
	int i = 0;
	for (q, i; q != p + n; q++, i++) {
		cout << "老师姓名：" << q->tea_name << endl;
		for (int j = 0; j < m; j++) {
			cout << "\t学生姓名：" << q->stu[j].stu_name << " 考试分数：" << q->stu[j].score << endl;
		}
		cout << endl;
	}
}
int main() {
	//创建结构体变量
	Tea tea[3];
	full(&tea[0], 5, 3);
	print(&tea[0], 5, 3);
}

6.9 结构体案例2

案例描述：

设计一个英雄的结构体，包括成员姓名，年龄，性别；创建结构体数组，数组中放入5名英雄。

将数组中的英雄按照年龄进行升序排序，最终打印出结果

代码实现：

#include <iostream>
using namespace std;
#include <string>

/*
设计一个英雄的结构体，包括成员姓名，年龄，性别；创建结构体数组，数组中放入5名英雄。

将数组中的英雄按照年龄进行升序排序，最终打印出结果*/

struct person {
	string name;
	int age;
	string sex;
};

//排序
void sort(person* p,int len) {
	for (int i = 0; i < len; i++) {
		//从已知有序元素中从后往前扫描，将无序中大的元素插入到有序元素中
		for (int j = i; j > 0 && p[j - 1].age > p[j].age; j--) {
			person temp = p[j - 1];
			p[j - 1] = p[j];
			p[j] = temp;
		}
	}
}
int main() {
	person per[5]{
		{"刘备",23,"男"},
		{"关羽",22,"男"},
		{"张飞",20,"男"},
		{"赵云",21,"男"},
		{"貂蝉",19,"女"}
	};
	for (person p : per) {
		cout << "姓名：" << p.name << " 年龄：" << p.age << " 性别：" << p.sex << endl;
	}
	cout << endl;
	sort(per, 5);
	for (person p : per) {
		cout << "姓名：" << p.name << " 年龄：" << p.age << " 性别：" << p.sex << endl;
	}
}

python对数组先排序后去重

排序用的sorted函数，去重自己实现的，当然去重也可以用set

import sys

num = int(sys.stdin.readline().strip("\n"))
list1=[]
for i in range(num):
    m=int(sys.stdin.readline().strip("\n"))
    list1.append(m)
    #排序
    list1=sorted(list1)
print(list1)
length =len(list1)
i=0
result=[]
#核心去重
while i < length-1:
    if list1[i] != list1[i+1]:
        result.append(list1[i])
    i=i+1
for i in result:
    print(i,end=' ')

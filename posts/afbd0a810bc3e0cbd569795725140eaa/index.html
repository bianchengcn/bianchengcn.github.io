moni.html

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Title</title>
    <link href="/css/moni.css"  type="text/css" rel="stylesheet">
</head>
<body>


<div class="a">

    <div class="b">
        <div class="c">注册</div>
    </div>

    <div class="e">
                <p>账号</p>
                <input type="text" placeholder="请输入用户名" name="username"/>
                <p>密码</p>
                <input type="password" placeholder="请输入密码"  name="password"/>
                <br/>
                <input type="submit" value="登录"/>
    </div>

</div>

</body>
</html>

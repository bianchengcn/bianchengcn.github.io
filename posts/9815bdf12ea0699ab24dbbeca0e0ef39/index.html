使用

进入build目录,创建demo.html

<!DOCTYPE html>
<html lang="zh">
<head>
<meta charset="UTF-8">
</head>
<script type="text/javascript" src="ckeditor.js"></script>
<script type="text/javascript" src="translations/zh-cn.js"></script>
<body>
<div id="editor"></div>
</body>
<script>
	    ClassicEditor
        .create(document.querySelector('#editor'), {
            language: {
                ui: 'zh-cn'
            }
        })
        .then(editor => {
            window.editor = editor;
        })
        .catch(error => {
            console.error(error);
        });
</script>
</html>

一个简单的编辑器创建完成.并且可以通过window.editor.getData()获取内容,
window.editor.setData()插入内容

配置图片上传

<script>
	    ClassicEditor
        .create(document.querySelector('#editor'), {
            language: {
                ui: 'zh-cn'
            },
            ckfinder: {
                // Upload the images to the server using the CKFinder QuickUpload command.
                uploadUrl: '上传接口',
                // Define the CKFinder configuration (if necessary).
                options: {
                    resourceType: 'Images'
                }
            }
        })
        .then(editor => {
            window.editor = editor;
        })
        .catch(error => {
            console.error(error);
        });
</script>

需要自己实现一个上传图片的接口,返回如下json格式的信息

{
	uploaded:1
	filename:"xxx.png"
	url:"http://xx"
}

编辑器会使用返回的url作为的src来创建图片

安装其它插件

进入ckeditor5-build-classic-master目录
打开命令行(git bash here)运行npm install
以base64图片插件为例

npm install --save @ckeditor/ckeditor5-upload

编辑src下的ckeditor.js

// 导入模块
import Base64UploadAdapter from '@ckeditor/ckeditor5-upload/src/adapters/base64uploadadapter';
// 添加插件
ClassicEditor.builtinPlugins = [
	其它插件,
	Base64UploadAdapter
];
// 默认配置(不用改)可以把语言直接改为中文
ClassicEditor.defaultConfig = {
	toolbar: {
		items: [
			'heading',
			'|',
			'bold',
			'italic',
			'link',
			'bulletedList',
			'numberedList',
			'|',
			'indent',
			'outdent',
			'|',
			'imageUpload',
			'blockQuote',
			'insertTable',
			'mediaEmbed',
			'undo',
			'redo'
		]
	},
	image: {
		toolbar: [
			'imageStyle:full',
			'imageStyle:side',
			'|',
			'imageTextAlternative'
		]
	},
	table: {
		contentToolbar: [
			'tableColumn',
			'tableRow',
			'mergeTableCells'
		]
	},
	// This value must be kept in sync with the language defined in webpack.config.js.
	language: 'en'
};

构建ckeditor

yarn run build

最后build目录下会生成新的ckeditor.js

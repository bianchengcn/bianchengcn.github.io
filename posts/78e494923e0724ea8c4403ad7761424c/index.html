df = pd.read_excel(“文件id”, usecols=[0, 5]) # 指定读取第1列和第6列
df = pd.read_excel(“文件id””, usecols=[0]) # 指定读取第1列
df = pd.read_excel(“文件id””, usecols=[0, 1，2]) # 指定读取第1列、第2列第三

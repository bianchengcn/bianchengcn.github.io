js数字排序

解题思路：数字=》字符串=》数组=》排序

let num = 1654

/*
	step one:数字转为字符串
	1.num.toString()
	2.String(num)
*/

let str = num.toString()

/*
	step two:字符串转为数组
	1.str.split('')
	2.[...str]
*/

let arr = str.split('')

/*
	step3:数组排序
	.sort()
*/

let newArr = arr.sort((a, b) => a - b)  //升序

/*
	step4:数组转为字符串
	.join('')
*/

let newNum = Number(arr.join(''))
console.log(newNum)  //1456

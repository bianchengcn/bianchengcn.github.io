pandas按照行执行自定义函数

表格信息如图
想得到第四列，是前三列的求和
上代码

import pandas as pd

# 读取表格信息
df = pd.DataFrame(pd.read_excel("样例表.xlsx", engine='openpyxl'))


# 自定义求和函数，传入  series
def sum_fun(series):
    aa = int(series["数据1"])
    bb = int(series["数据2"])
    cc = int(series["数据3"])
    dd = aa + bb + cc
    return dd


# 执行函数，并在表格中添加新列 求和 将每行的计算的结果写入最后一列
df['求和'] = df.apply(sum_fun, axis=1)

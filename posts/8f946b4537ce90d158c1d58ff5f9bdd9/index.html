C#基础 string PadLeft 在字符串前、后加入指定字符，达到指定长度

  •  .NET Framework : 4.7.2
  •        IDE : Visual Studio Community 2019
  •         OS : Windows 10 x64
  •     typesetting : Markdown
  •         blog : xinshaopu.blog.csdn.net

 

code

using System;

namespace ConsoleApp
{
    class Program
    {
        static void Main(string[] args)
        {

            string str = "1  i am a student  1";

            // 只会删除字符串首字母前面的空格与尾字母后面的空格
            Console.WriteLine(str.Trim());

            // 在str字符串前加入空格，使得字符串达到指定长度
            Console.WriteLine(str.PadLeft(25));

            // 在str字符串前加入a，使得字符串达到指定长度
            Console.WriteLine(str.PadLeft(25, 'a'));

            Console.WriteLine(str.PadRight(25, 'a'));

            Console.ReadKey();
        }
    }
}

 

result

1  i am a student  1
     1  i am a student  1
aaaaa1  i am a student  1
1  i am a student  1aaaaa

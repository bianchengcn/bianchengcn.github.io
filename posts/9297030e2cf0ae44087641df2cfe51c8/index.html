25. K 个一组翻转链表(Java)，详细注释

给你链表的头节点 head ，每 k 个节点一组进行翻转，请你返回修改后的链表。

k 是一个正整数，它的值小于或等于链表的长度。如果节点总数不是 k 的整数倍，那么请将最后剩余的节点保持原有顺序。

你不能只是单纯的改变节点内部的值，而是需要实际进行节点交换。

示例 1：

输入：head = [1,2,3,4,5], k = 2
输出：[2,1,4,3,5]

示例 2：

输入：head = [1,2,3,4,5], k = 3
输出：[3,2,1,4,5]

图解思路：

代码：

/**
 * Definition for singly-linked list.
 * public class ListNode {
 *     int val;
 *     ListNode next;
 *     ListNode() {}
 *     ListNode(int val) { this.val = val; }
 *     ListNode(int val, ListNode next) { this.val = val; this.next = next; }
 * }
 */
class Solution {
    public ListNode reverseKGroup(ListNode head, int k) {
        // 创建大链表的伪头节点
        ListNode dummy = new ListNode(0);
        dummy.next = head;

        // 用于记录需要翻转子链表的伪头节点和尾节点
        ListNode pre = dummy;
        ListNode end = dummy;

        while(end.next != null){
            //end走到需要翻转子链表尾节点的位置
            for (int i = 0; i < k && end != null; i++) end = end.next;
            if (end == null) break;
            // 用于记录需要翻转链表的头节点和下一组的头节点
            ListNode start = pre.next;
            ListNode next = end.next;
            // 把需要翻转的子链表从大链表中断开
            end.next = null;
            // 对子链表进行翻转，把伪头节点连接到翻转后子链表的头节点
            pre.next = reverseList(start);
            // 此时star已成为子链表的尾节点，此步把子链表连接回大链表
            start.next = next;
            // 伪节点的指针转移到新准备要翻转子链表的前一个节点的位置
            pre = start;
            // 翻转后的end已经成为子链表的头节点，重新把它恢复到与伪头节点相同位置
            end = pre;
        }

        return dummy.next;
    }

    public ListNode reverseList(ListNode head) {
        ListNode pre = null;
        ListNode cur = head;
        ListNode temp = null;
        while(cur != null){
            // 记录当前节点的下一个节点
            temp = cur.next;
            // 将当前节点指向前pre
            cur.next = pre;
            // pre和cur节点前进一位
            pre = cur;
            cur = temp;
        }
        return pre;

    }
}

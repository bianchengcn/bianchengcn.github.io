node-rsa 公钥解密

const NodeRSA = require('node-rsa');
const key = new NodeRSA('-----BEGIN PUBLIC KEY-----\n'+
...........
'-----END PUBLIC KEY-----');

const token = "token";
const decrypted = key.decryptPublic(token, 'utf8');
console.log('decrypted: ', decrypted);

Functionhextointer(strin)

Dimi,j,k,result

result=0

Fori=1ToLen(strin)

IfMid(strin,i,1)="f"OrMid(strin,i,1)="F"Then

j=15

EndIf

IfMid(strin,i,1)="e"OrMid(strin,i,1)="E"Then

j=14

EndIf

IfMid(strin,i,1)="d"OrMid(strin,i,1)="D"Then

j=13

EndIf

j=12

EndIf

Next

js中函数的length属性

前端面试题：函数的length长度了解 

` <script>
    // 函数的length属性
    function a(a1, b2, c3) { }//length长度为形参的个数
    console.log('函数a的length长度', a.length);
    function b(a1, b2 = 0, c3) { } // 当有默认值时，取默认值前的个数
    console.log('函数b的length长度', b.length);
    function c(a1, b2, ...args) { } // 不统计剩余参数
    console.log('函数c的length长度', c.length);
  </script>`
